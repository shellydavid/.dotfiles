
------- Log -------
glo - git log
glod - git log with diffs

------- Branches -------
gb - git branch
gbd - git branch --delete
gbD - git branch --delete --force
gco - git checkout
gcb - git checkout -b
gswm - git switch main
gswd - git switch development

------- Commits -------
gs - git status
ga - git add
gcm - git commit -m
gd - git diff
gp - git push
gpu - git pull
ggpull - git pull origin <current_branch>
ggpush - git push origin <current_branch>
grb - git rebase
gcp - git cherry-pick
gcpa - git cherry-pick --abort
gcpc - git cherry-pick--continue
oops - git reset --soft HEAD~1

------- Stash -------
gst - git stash (includes untracked)
gstl - git stash list
gstp - git stash pop
gsts - git stash show
gstd - git stash drop
